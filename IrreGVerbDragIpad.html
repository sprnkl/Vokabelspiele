document.querySelectorAll('.verb').forEach((verb) => { // Für Maus-Drag-&-Drop verb.addEventListener('dragstart', (event) => { event.dataTransfer.setData('text/plain', ''); // Für Desktop-Browser verb.classList.add('dragging'); }); verb.addEventListener('dragend', () => { verb.classList.remove('dragging'); }); // Für Touch-Drag-&-Drop verb.addEventListener('touchstart', (event) => { const touch = event.touches[0]; verb.classList.add('dragging'); verb.style.position = 'absolute'; verb.style.left = `${touch.pageX - verb.offsetWidth / 2}px`; verb.style.top = `${touch.pageY - verb.offsetHeight / 2}px`; }); document.addEventListener('touchmove', (event) => { const dragged = document.querySelector('.dragging'); if (!dragged) return; const touch = event.touches[0]; dragged.style.left = `${touch.pageX - dragged.offsetWidth / 2}px`; dragged.style.top = `${touch.pageY - dragged.offsetHeight / 2}px`; }); document.addEventListener('touchend', () => { const dragged = document.querySelector('.dragging'); if (dragged) { dragged.classList.remove('dragging'); dragged.style.position = ''; dragged.style.left = ''; dragged.style.top = ''; } }); }); document.querySelectorAll('.drop-zone').forEach((zone) => { // Für Maus-Drag-&-Drop zone.addEventListener('dragover', (event) => { event.preventDefault(); }); zone.addEventListener('drop', (event) => { event.preventDefault(); const dragged = document.querySelector('.dragging'); if (dragged && dragged.dataset.match === zone.dataset.match) { zone.textContent = dragged.textContent; zone.classList.add('correct'); dragged.classList.add('hidden'); } }); // Für Touch-Drag-&-Drop zone.addEventListener('touchend', () => { const dragged = document.querySelector('.dragging'); if (dragged && dragged.dataset.match === zone.dataset.match) { zone.textContent = dragged.textContent; zone.classList.add('correct'); dragged.classList.add('hidden'); } }); });
